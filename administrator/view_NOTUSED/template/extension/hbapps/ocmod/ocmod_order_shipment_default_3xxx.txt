                <search><![CDATA[$data['histories'] = array();]]></search>
                <add position="before"><![CDATA[

				$this->language->load('extension/module/order_shipment');
				$data['column_service']     = $this->language->get('column_service');
				$data['column_code']        = $this->language->get('column_code');
				$data['column_link']        = $this->language->get('column_link');
				$data['text_courier']       = $this->language->get('text_courier');
				$data['column_delivery_date']       = $this->language->get('column_delivery_date');
				$data['column_shipped_quantity']       = $this->language->get('column_shipped_quantity');
				
				$data['order_shipment_data'] = array();
				
				$order_shipment_data = $this->db->query("SELECT *, a.id as shipment_id, c.name as pname, b.name as cname FROM " . DB_PREFIX . "hb_shipment_order_info a , `" . DB_PREFIX . "hb_shipping_company` b, " . DB_PREFIX . "order_product c where a.courier_id = b.id and a.order_product_id = c.order_product_id and a.order_id = c.order_id and a.order_id = '".(int)$this->request->get['order_id']."'");
	
        		if ($order_shipment_data->rows) {	
					$order_shipments = 	$order_shipment_data->rows;	
					$query_count = $this->db->query("SELECT count(*) as count FROM `" . DB_PREFIX . "hb_shipment_order_info` WHERE order_id = '".(int)$order_id."' AND `code` = (SELECT code FROM " . DB_PREFIX . "hb_shipment_order_info WHERE order_id = '".(int)$order_id."' LIMIT 1)");
					if ($query_count->row['count'] == count($order_shipments)) {
						$data['bulk'] = 1;
					}else{
						$data['bulk'] = 0;
					}
					
        			foreach ($order_shipments as $row) {
						$option_data = array();

						$options = $this->model_account_order->getOrderOptions($this->request->get['order_id'], $row['order_product_id']);
		
						foreach ($options as $option) {
							if ($option['type'] != 'file') {
								$value = $option['value'];
							} else {
								$upload_info = $this->model_tool_upload->getUploadByCode($option['value']);
		
								if ($upload_info) {
									$value = $upload_info['name'];
								} else {
									$value = '';
								}
							}
		
							$option_data[] = array(
								'name'  => $option['name'],
								'value' => (utf8_strlen($value) > 20 ? utf8_substr($value, 0, 20) . '..' : $value)
							);
						}
						
        				//$data['bulk'] = $row['bulk'];
        				$data['partner_name'] = $row['cname'];
        				$data['tracking_number'] = $row['code'];
        				$data['tracking_link'] = str_replace('{tracking_id}',$row['code'],$row['link']);
        				$data['delivery_date'] = ($row['delivery_date'] == NULL or $row['delivery_date'] == '0000-00-00') ? '' : date('d-m-Y',strtotime($row['delivery_date']));
        				
        				$data['order_shipment_data'][] = array(
        					'shipment_id'	=> $row['shipment_id'],
        					'product_id'	=> $row['product_id'],
        					'product_name'	=> $row['pname'],
							'option'   		=> $option_data,
							'model'			=> $row['model'],
							'shipped_qty'	=> $row['shipped_qty'],
        					'partner_name'	=> $row['cname'],
        					'code'			=> $row['code'],
        					'tracking_link'	=> str_replace('{tracking_id}',$row['code'],$row['link']),
        					'delivery_date'	=> ($row['delivery_date'] == NULL or $row['delivery_date'] == '0000-00-00') ? '' : date('d-m-Y',strtotime($row['delivery_date'])) ,
        					'mail'			=> $row['mail']
        				);
        			}
        		}
				
				]]>
                </add>
